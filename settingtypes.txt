
[General]

# Make leaves change color in the fall.
mymonths_leaves (Autumn Leaves) bool true

# Puddles appear when raining.
mymonths_use_puddles (Puddles) bool true

# Have snow accumulate on the ground.
mymonths_snow_on_ground (Snow on Ground) bool true

# Turn weather on or off.
mymonths_use_weather (Weather) bool true

# This will make storms and hail cause damage.
mymonths_damage (Weather Damage) bool false
